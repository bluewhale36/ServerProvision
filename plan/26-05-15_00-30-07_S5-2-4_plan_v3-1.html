Java 직렬화

// global/trash/PurgeLogDetails.java — sealed interface + 2 record
public sealed interface PurgeLogDetails permits SuccessDetails, FailedDetails {}

public record SuccessDetails(
    String triggeredBy, String trashedPath, String originalPath,
    String manifestHash, String markerSignature, int attemptCount
) implements PurgeLogDetails {}

public record FailedDetails(
    String triggeredBy, int attemptNumber, int tickAttemptCount,
    String exceptionClass, String failureReason, String stackHash
) implements PurgeLogDetails {}

@Converter
public class PurgeLogDetailsConverter implements AttributeConverter<PurgeLogDetails, String> {

    @Override
    public String convertToDatabaseColumn(PurgeLogDetails attr) {
        return attr == null ? null : mapper.writeValueAsString(attr);
    }

    @Override
    public PurgeLogDetails convertToEntityAttribute(String dbData) {
        // outcome 컬럼은 entity 단에서 이미 읽혔으므로 polymorphic deserialize.
        // @JsonTypeInfo / @JsonSubTypes 로 type discriminator. 또는 outer entity 가 outcome 보고 분기.
    }
}
JSON 직렬화 전략 sealed interface + Jackson @JsonTypeInfo. entity 의 outcome 컬럼이 discriminator 역할 — converter 는 outcome 을 받아 적절한 record 로 deserialize. tools.jackson.* 사용 (CLAUDE.md §기술 스택).

스키마 확장 시 절차

  • 새 키 추가 — record 에 컴포넌트 추가 + 기존 row 는 missing → default 처리 (Jackson @JsonProperty(defaultValue=...))
  • 키 제거 — record 컴포넌트 제거 + 직렬화에서 무시. 기존 row 의 잉여 키는 Jackson FAIL_ON_UNKNOWN_PROPERTIES=false 로 흡수
  • schema migration 0 — 모든 변경이 application 단에서만 (DB JSON 타입 그대로)
🔎 4. 주요 쿼리 패턴

Q1. 자원 단위 회고 (운영자 페이지)

SELECT id, origin, outcome, occurred_at, purged_at, details
FROM purge_log
WHERE resource_type = 'OS_ISO' AND resource_id = 27
ORDER BY occurred_at ASC;

활용 인덱스 : idx_purge_log_resource — covering range scan

Q2. 다음 cron 의 재시도 후보 자동 식별 (별 SPI 없이)

-- "재시도 후보 조회 SPI" 자체를 제거 (사용자 결정).
-- TrashTtlWorker 는 그냥 휴지통의 TTL 만료 자원을 순회.
-- 자원이 휴지통에 남아있다 = 아직 SUCCESS 안 했다 → 자연스러운 다음 cron 후보.

SELECT resource_id
FROM iso          -- 도메인별 자원 테이블
WHERE is_deleted = true
  AND trashed_at <= NOW() - INTERVAL :ttl_days DAY;
-- 이미 SUCCESS 한 자원은 entity hard-delete 로 row 자체가 사라졌으므로 자연 제외

Q3. 자원별 누적 FAILED 횟수 (attemptNumber 계산용)

SELECT COUNT(*) FROM purge_log
WHERE resource_type = :type
  AND resource_id   = :id
  AND outcome       = 'FAILED';

활용 인덱스 : idx_purge_log_resource + outcome 컬럼 후 분기 — MariaDB 가 적절히 사용

Q4. 마지막 outcome=FAILED 자원 목록 (재시도 대기 배지)

-- 자원별 마지막 행이 FAILED 인 자원 셋
WITH last_log AS (
  SELECT resource_type, resource_id,
         ROW_NUMBER() OVER (PARTITION BY resource_type, resource_id ORDER BY occurred_at DESC) AS rn,
         outcome
  FROM purge_log
)
SELECT resource_type, resource_id
FROM last_log
WHERE rn = 1 AND outcome = 'FAILED';

MariaDB 10.2+ 의 window function 활용. 또는 자원 단위 GROUP BY + HAVING MAX(occurred_at) 패턴.

Q5. Pageable 조회 (관리자 purge-log 페이지)

SELECT * FROM purge_log
WHERE (:resource_type IS NULL OR resource_type = :resource_type)
  AND (:origin        IS NULL OR origin        = :origin)
  AND (:outcome       IS NULL OR outcome       = :outcome)
  AND occurred_at BETWEEN :from AND :to
ORDER BY occurred_at DESC
LIMIT :size OFFSET :offset;

활용 인덱스 : idx_purge_log_outcome_occurred (outcome 필터 + 정렬 동시) 또는 idx_purge_log_resource

Q6. 자원 displayName 검색 (JSON parse 없음 — v3-1 의 효과)

SELECT * FROM purge_log
WHERE display_name LIKE '%Rocky Linux%'
ORDER BY occurred_at DESC;

v3 는 JSON parse 필요했음 (JSON_EXTRACT(details, '$.displayName') LIKE ...) — v3-1 의 컬럼화로 단순 LIKE 사용 가능. 자원 검색 빈도가 높으면 idx_purge_log_display_name(display_name) 추가 가능 (선택).

⚙ 5. JPA 매핑 & 인덱스

TrashSettings entity

@Entity
@Table(name = "trash_settings")
@Getter
@NoArgsConstructor(access = AccessLevel.PROTECTED)
public class TrashSettings extends BaseTimeEntity {

    @Id
    private Long id;  // 항상 1L

    @Column(nullable = false) private int ttlDays;
    @Column(nullable = false) private boolean autoPurgeEnabled;
    @Column(nullable = false, length = 64) private String purgeCronExpression;
    @Column(nullable = false, length = 64) private String notifyCronExpression;
    @Column(nullable = false, length = 64) private String notifyDaysBefore;
    @Column(nullable = false, length = 128) private String notificationChannels;
    @Column(nullable = false) private int retryMaxAttempts;
    @Column(nullable = false) private long retryBackoffBaseMs;

    public static TrashSettings defaults() {
        // id=1L + 모든 default 값 채움 빌더
    }

    public Duration getTtl() { return Duration.ofDays(ttlDays); }
    public List<Integer> getNotifyDaysBeforeList() {
        return Arrays.stream(notifyDaysBefore.split(",")).map(String::trim).map(Integer::parseInt).toList();
    }
    public Set<NotifyChannel> getNotificationChannelSet() {
        return Arrays.stream(notificationChannels.split(",")).map(NotifyChannel::valueOf).collect(toSet());
    }
}

PurgeLog entity

@Entity
@Table(
    name = "purge_log",
    indexes = {
        @Index(name = "idx_purge_log_resource",          columnList = "resource_type, resource_id, occurred_at DESC"),
        @Index(name = "idx_purge_log_outcome_occurred",  columnList = "outcome, occurred_at DESC"),
        @Index(name = "idx_purge_log_origin",            columnList = "origin")
    }

@Getter
@NoArgsConstructor(access = AccessLevel.PROTECTED)
public class PurgeLog {

    @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false, length = 32)
    private ResourceType resourceType;

    @Column(nullable = false)
    private Long resourceId;

    @Column(nullable = false, length = 256)
    private String displayName;  // v3-1 발탁 컬럼

    @Enumerated(EnumType.STRING)
    @Column(nullable = false, length = 16)
    private PurgeOrigin origin;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false, length = 8)
    private PurgeOutcome outcome;

    @Column(nullable = false, columnDefinition = "TIMESTAMP(6)")

    @Column(columnDefinition = "TIMESTAMP(6)")

    @Convert(converter = PurgeLogDetailsConverter.class)
    @Column(nullable = false, columnDefinition = "JSON")
    private PurgeLogDetails details;

    /* 도메인 메서드 — outcome 다형성을 record + sealed 로 표현 */
    public static PurgeLog success(PurgeRequest req, Markable snapshot, int tickAttemptCount,
                                   Instant occurredAt, Instant purgedAt) { /* ... */ }

    public static PurgeLog failure(PurgeRequest req, Throwable lastError,
                                   int attemptNumber, int tickAttemptCount,
                                   Instant occurredAt) { /* purgedAt=null */ }
}

BaseTimeEntity 사용 여부

PurgeLog 은 BaseTimeEntity 를 상속하지 않는다. 이유 : created_at / modified_at 보다 occurred_at / purged_at 가 도메인 의미를 직접 표현. base 의 자동 컬럼이 노이즈가 됨. INSERT only 라 modified_at 도 무의미.

SHOW CREATE TABLE 예상 출력

CREATE TABLE `purge_log` (
  `id`            BIGINT NOT NULL AUTO_INCREMENT,
  `resource_type` VARCHAR(32)  NOT NULL,
  `resource_id`   BIGINT       NOT NULL,
  `display_name`  VARCHAR(256) NOT NULL,
  `origin`        VARCHAR(16)  NOT NULL,
  `outcome`       VARCHAR(8)   NOT NULL,
  `occurred_at`   TIMESTAMP(6) NOT NULL,
  `purged_at`     TIMESTAMP(6) NULL,
  `details`       JSON         NOT NULL,
  PRIMARY KEY (`id`),
  KEY `idx_purge_log_resource`         (`resource_type`, `resource_id`, `occurred_at`),
  KEY `idx_purge_log_outcome_occurred` (`outcome`, `occurred_at`),
  KEY `idx_purge_log_origin`           (`origin`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
MariaDB JSON 컬럼 주의 MariaDB 의 JSON 은 사실상 LONGTEXT alias (validation 만 적용). JSON 함수 인덱스는 생성 불가 — 필요 시 generated column 후 인덱스 패턴. 본 슬라이스는 JSON 키 기반 검색 빈도가 낮으므로 별도 처리 안 함. 운영 병목 시 PostgreSQL jsonb + GIN 이관 검토.

6. PurgeExecutor 흐름 (v3 와 동일, 컬럼 분리 반영) 
public PurgeResult execute(PurgeRequest req) {
    Instant occurredAt = Instant.now();
    Markable snapshot = scanner.findTrashedById(req.resourceId()).orElseThrow();
    String displayName = snapshot.displayName();  // v3-1 — 컬럼 저장용 발탁

    Exception lastFailure = null;
    int maxAttempts = req.origin().retriesAllowed() ? settings.getRetryMaxAttempts() : 1;

    for (int tickAttempt = 1; tickAttempt <= maxAttempts; tickAttempt++) {
        try {
            scanner.purgeFromTrash(req.resourceId());
            Instant purgedAt = Instant.now();
            SuccessDetails details = new SuccessDetails(
                req.triggeredBy(),
                snapshot.getTrashedPathOrNull(), snapshot.getResourcePath().toString(),
                snapshot.getManifestHash(), snapshot.getMarkerSignature(),
                tickAttempt
            );
            purgeLogRepository.save(PurgeLog.success(
                req.resourceType(), req.resourceId(), displayName,
                req.origin(), occurredAt, purgedAt, details));
            return PurgeResult.success(req);

        } catch (PurgeIoFailedException ex) {
            lastFailure = ex;
            if (tickAttempt < maxAttempts) {
                Thread.sleep(backoff(tickAttempt));  // 1s/2s/4s
            }
        }
    }

    int historicalFailedCount = purgeLogRepository.countByResourceAndOutcome(
        req.resourceType(), req.resourceId(), PurgeOutcome.FAILED);
    FailedDetails details = new FailedDetails(
        req.triggeredBy(),
        historicalFailedCount + 1, maxAttempts,
        lastFailure.getClass().getSimpleName(), lastFailure.getMessage(),
        stackHashOf(lastFailure)
    );
    purgeLogRepository.save(PurgeLog.failure(
        req.resourceType(), req.resourceId(), displayName,
        req.origin(), occurredAt, details));
    return PurgeResult.failed(req, lastFailure);
}
