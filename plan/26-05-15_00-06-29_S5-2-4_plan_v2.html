3.2 PurgeExecutor 단일 진입점 (v2)

// global/trash/PurgeExecutor.java (신규)
public class PurgeExecutor {
    public PurgeResult execute(PurgeRequest req) {
        // 1) audit_log 의 사전 row 안 만든다 — 성공 시점에만 insert (실패는 attempt 에)
        // 2) scanner.purgeFromTrash(id) 호출 with retry (TTL_AUTO 만)
        for (int attempt = 1; attempt <= retryPolicy.maxAttempts(); attempt++) {
            try {
                Markable snapshot = scanner.findTrashedById(req.resourceId()).orElseThrow();
                scanner.purgeFromTrash(req.resourceId());
                purgeAuditLogRepository.save(toAuditLog(req, snapshot));
                return PurgeResult.success(req);
            } catch (PurgeIoFailedException ex) {
                purgeAttemptRepository.save(toAttempt(req, attempt, ex));
                if (!req.origin().retriesAllowed() || attempt == retryPolicy.maxAttempts()) {
                    return PurgeResult.failed(req, attempt, ex);
                }
                Thread.sleep(retryPolicy.backoffFor(attempt));  // 1s/2s/4s
            }
        }
        // unreachable
    }
}
